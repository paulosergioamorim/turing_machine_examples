0 * * l 0
0 _ # l escreve0

;      começa escrevendo 0 antes do espaço em branco
escreve0 * 0 * gotoend

;      percorrer até o ultimo caracter diferente de 0 ou 1
gotoend * * r *

gotoend u * l escrevenum
gotoend z * l escrevenum

;      escreve z para zero e u para 1

;      incrementa um no numero antes do # apos percorrer até o ultimo digito que nao foi especificado

inc1 0 1 * gotoend
inc1 1 0 l propaga

;      propaga o incrementador para tras

propaga * 1 * gotoend
propaga 1 0 l propaga
